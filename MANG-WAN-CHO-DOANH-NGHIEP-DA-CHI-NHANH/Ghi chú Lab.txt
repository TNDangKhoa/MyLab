FW

config system interface
    edit "port1"
	set mode static
        set ip 192.168.10.2 255.255.255.252
        set allowaccess ping
    next
    edit "port2"
	set mode static
        set ip 192.168.1.1 255.255.255.0
        set allowaccess ping
    next
end

config router static
    edit 1
        set gateway 192.168.10.1  # IP cổng e2/1 của R10
        set device "port1"
    next
end

config firewall policy
    edit 1
        set srcintf "port2"
        set dstintf "port1"
        set srcaddr "all"
        set dstaddr "all"
        set action accept
        set schedule "always"
        set service "ICMP"
    next
end
